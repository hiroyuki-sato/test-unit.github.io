def long_display
  if location.size == 1
    location_display = location[0].sub(/\A(.+:\d+).*/, ' [\\1]')
  else
    location_display = "\n    [#{location.join("\n     ")}]"
  end
  "#{label}:\n#@test_name#{location_display}:\n#@message"
end
